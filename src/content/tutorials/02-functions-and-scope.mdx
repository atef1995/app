## What are Functions?

Functions are reusable blocks of code that perform specific tasks. Think of them as mini-programs within your program!

```javascript
function greetUser(name) {
  return `Hello, ${name}! Welcome to Vibed to Cracked!`;
}

// Using the function
let message = greetUser("Alex");
console.log(message); // "Hello, Alex! Welcome to Vibed to Cracked!"
```

## Function Declaration vs Function Expression

### 1. Function Declaration

```javascript
function calculateArea(width, height) {
  return width * height;
}
```

**Features:**

- Hoisted (can be called before declaration)
- Named function
- Traditional syntax

### 2. Function Expression

```javascript
const calculateArea = function (width, height) {
  return width * height;
};
```

**Features:**

- Not hoisted
- Can be anonymous
- Stored in a variable

### 3. Arrow Functions (Modern JavaScript)

```javascript
const calculateArea = (width, height) => {
  return width * height;
};

// Or even shorter for single expressions:
const calculateArea = (width, height) => width * height;
```

**Features:**

- Concise syntax
- No `this` binding (advanced topic)
- Perfect for short functions

## Parameters and Arguments

### Basic Parameters

```javascript
function createProfile(name, age, city) {
  return {
    name: name,
    age: age,
    city: city,
    isActive: true,
  };
}

let profile = createProfile("Sarah", 28, "New York");
```

### Default Parameters

```javascript
function greetUser(name = "Friend", timeOfDay = "day") {
  return `Good ${timeOfDay}, ${name}!`;
}

console.log(greetUser()); // "Good day, Friend!"
console.log(greetUser("Alice")); // "Good day, Alice!"
console.log(greetUser("Bob", "morning")); // "Good morning, Bob!"
```

### Rest Parameters (...)

```javascript
function calculateTotal(...numbers) {
  let total = 0;
  for (let number of numbers) {
    total += number;
  }
  return total;
}

console.log(calculateTotal(1, 2, 3, 4, 5)); // 15
console.log(calculateTotal(10, 20)); // 30
```

## Return Values

Functions can return values using the `return` statement:

```javascript
function multiply(a, b) {
  return a * b; // Returns the result
}

function isEven(number) {
  return number % 2 === 0; // Returns true or false
}

function processData(data) {
  // Multiple return paths
  if (!data) {
    return "No data provided";
  }

  if (data.length === 0) {
    return "Empty data set";
  }

  return `Processed ${data.length} items`;
}
```

**Important:** Functions without a `return` statement return `undefined`.

## Scope: Where Variables Live

Scope determines where variables can be accessed in your code.

### Global Scope

```javascript
let globalVariable = "I'm accessible everywhere!";

function testFunction() {
  console.log(globalVariable); // âœ… Works fine
}

console.log(globalVariable); // âœ… Works fine
```

### Function Scope

```javascript
function myFunction() {
  let functionVariable = "I only exist inside this function";
  console.log(functionVariable); // âœ… Works fine
}

// console.log(functionVariable); // âŒ ReferenceError!
```

### Block Scope (let and const)

```javascript
if (true) {
  let blockVariable = "I only exist in this block";
  const anotherBlockVariable = "Me too!";
  var functionScoped = "I'm different...";
}

// console.log(blockVariable); // âŒ ReferenceError!
// console.log(anotherBlockVariable); // âŒ ReferenceError!
console.log(functionScoped); // âœ… Works (var is function-scoped)
```

### Scope Chain

```javascript
let outerVariable = "I'm outside";

function outerFunction() {
  let middleVariable = "I'm in the middle";

  function innerFunction() {
    let innerVariable = "I'm inside";

    // Can access all three!
    console.log(outerVariable); // âœ… "I'm outside"
    console.log(middleVariable); // âœ… "I'm in the middle"
    console.log(innerVariable); // âœ… "I'm inside"
  }

  innerFunction();
}

outerFunction();
```

## Real-World Examples

### 1. User Input Validation

```javascript
function validateEmail(email) {
  if (!email) {
    return false;
  }

  return email.includes("@") && email.includes(".");
}

function validatePassword(password) {
  return password && password.length >= 8;
}

function validateUser(email, password) {
  const isEmailValid = validateEmail(email);
  const isPasswordValid = validatePassword(password);

  return isEmailValid && isPasswordValid;
}
```

### 2. Data Processing

```javascript
function formatCurrency(amount, currency = "USD") {
  return `${currency} $${amount.toFixed(2)}`;
}

function calculateDiscount(price, discountPercent) {
  const discount = price * (discountPercent / 100);
  return price - discount;
}

function processPurchase(items) {
  let total = 0;

  items.forEach((item) => {
    const discountedPrice = calculateDiscount(item.price, item.discount || 0);
    total += discountedPrice;
  });

  return formatCurrency(total);
}

// Usage
const cartItems = [
  { name: "Laptop", price: 999, discount: 10 },
  { name: "Mouse", price: 29 },
  { name: "Keyboard", price: 79, discount: 5 },
];

console.log(processPurchase(cartItems)); // "USD $931.05"
```

### 3. Higher-Order Functions

```javascript
function createMultiplier(factor) {
  return function (number) {
    return number * factor;
  };
}

const double = createMultiplier(2);
const triple = createMultiplier(3);

console.log(double(5)); // 10
console.log(triple(4)); // 12
```

## Best Practices

### 1. Use Descriptive Names

```javascript
// âŒ Bad
function calc(x, y) {
  return x * y;
}

// âœ… Good
function calculateRectangleArea(width, height) {
  return width * height;
}
```

### 2. Keep Functions Small and Focused

```javascript
// âŒ Bad - does too many things
function processUserData(userData) {
  // validate data
  // format data
  // save to database
  // send email
  // update UI
  // ... 50 lines of code
}

// âœ… Good - single responsibility
function validateUserData(userData) {
  // validation logic only
}

function formatUserData(userData) {
  // formatting logic only
}

function saveUser(userData) {
  // database logic only
}
```

### 3. Use Arrow Functions for Short Operations

```javascript
// âœ… Perfect for array methods
const numbers = [1, 2, 3, 4, 5];
const doubled = numbers.map((n) => n * 2);
const evens = numbers.filter((n) => n % 2 === 0);

// âœ… Good for simple calculations
const calculateTax = (amount, rate) => amount * rate;
```

### 4. Avoid Global Variables

```javascript
// âŒ Bad - global pollution
let counter = 0;

function increment() {
  counter++;
}

// âœ… Good - encapsulated
function createCounter() {
  let counter = 0;

  return {
    increment: () => ++counter,
    decrement: () => --counter,
    getValue: () => counter,
  };
}

const myCounter = createCounter();
```

## Common Pitfalls to Avoid

### 1. Forgetting to Return

```javascript
// âŒ Bad
function addNumbers(a, b) {
  a + b; // Missing return!
}

console.log(addNumbers(2, 3)); // undefined

// âœ… Good
function addNumbers(a, b) {
  return a + b;
}
```

### 2. Modifying Parameters Directly

```javascript
// âŒ Bad - modifies original array
function addItem(array, item) {
  array.push(item);
  return array;
}

// âœ… Good - creates new array
function addItem(array, item) {
  return [...array, item];
}
```

### 3. Not Handling Edge Cases

```javascript
// âŒ Bad
function divide(a, b) {
  return a / b;
}

// âœ… Good
function divide(a, b) {
  if (b === 0) {
    return "Cannot divide by zero";
  }
  return a / b;
}
```
